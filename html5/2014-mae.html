Contacts API

        var contactData = {givenName: ["John"], familyName: ["Doe"]};
        var person = new mozContact(contactData);
        // for compatibility with FxOS 1.2 & earlier
        person.init && person.init(contactData);
        var saving = navigator.mozContacts.save(person);
        saving.onsuccess = ...;
        saving.onerror = ...;
